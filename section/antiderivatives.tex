\section{Antiderivatives}
\label{s:antiderivs}

\begin{definition}
  \label{def:antideriv}
  Let $f$ and $F$ be functions which are continuous on an interval $I = \closedint{a, b}$.
  Then $F$ is an \emph{antiderivative of $f$ on $I$} if $\D{F}{x} = f \pbrac{x}$ on $I$.
\end{definition}

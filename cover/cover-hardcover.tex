\documentclass{article}

\usepackage{calcnotes-cover}

% Obtain these dimensions from the printer
\newlength{\coverheight}
\setlength{\coverheight}{12.5in}

\newlength{\spinewidth}
\setlength{\spinewidth}{0.4in}

\newlength{\coverpagewidth}
\setlength{\coverpagewidth}{9.1in}

\newlength{\coverbleed}
\setlength{\coverbleed}{0.75in}

\newlength{\covermargin}
\setlength{\covermargin}{1.0in}

\newlength{\coverwidth}
\setlength{\coverwidth}{2\coverpagewidth+\spinewidth}

\newlength{\coverpageblockwidth}
\setlength{\coverpageblockwidth}{\coverpagewidth-\coverbleed-\covermargin}

\newlength{\coverpageblockheight}
\setlength{\coverpageblockheight}{\coverheight-\coverbleed-\covermargin}

\geometry{
  margin=0pt,
  paperwidth=\coverwidth,
  paperheight=\coverheight,
}


\begin{document}
\vspace*{\stretch{1}}
\hspace*{\stretch{1}}
\begin{minipage}[c][\coverpageblockheight][c]{\coverpageblockwidth}
  \input{cover-back}
\end{minipage}
% Spine
\begin{minipage}[c][\coverpageblockheight]{\spinewidth}
  \input{cover-spine}
\end{minipage}
% Front page
\begin{minipage}[c][\coverpageblockheight][c]{\coverpageblockwidth}
  \hfill
  \input{cover-front}
\end{minipage}
\hspace*{\stretch{1}}
\vspace*{\stretch{1}}
\end{document}

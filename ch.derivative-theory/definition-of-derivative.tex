\section{The definition of the derivative}
\label{sec:derivdefn}

\begin{definition}
  \label{def:deriv}
  Let $f$ be continuous at $x = a$.
  Then the \emph{derivative of $f$ at $a$} is the number
  \begin{equation}
    \label{eq:derivdef}
    f'(a) = \lim_{\Delta x \to 0} \frac{f \pbrac*{a + \Delta x} - f \pbrac{a}}{\Delta x}.
  \end{equation}
\end{definition}
